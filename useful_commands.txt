# Add delay
sudo tc qdisc add dev enp39s0 root netem delay 20ms

# Add packet loss
sudo tc qdisc add dev enp39s0 root netem loss 0.01%

# Limiting bandwidth to a particular limit
tc qdisc add dev enp39s0 root tbf rate 10gbit burst 10mb latency 50ms

# Show existing restrictions on enX0 (replace with needed interface)
tc qdisc show dev enp39s0 

# Remove all existing retrictions to that interface
sudo tc qdisc del dev enp39s0 root

# Changing rmem and wmem
sudo sysctl -w net.core.wmem_default=1073741824
sudo sysctl -w net.core.wmem_max=1073741824

sudo sysctl -w net.core.rmem_default=1073741824
